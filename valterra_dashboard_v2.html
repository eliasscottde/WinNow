#!/usr/bin/env python3
"""
Valterra — IEX DAM/RTM Adaptive Z-Score Strategy
Validated: Jan 2023 – Mar 2026 (975 trading days, 1,878 trades, 3 OOS years)
"""

import numpy as np
import pandas as pd
from datetime import datetime
import sqlite3, logging, schedule, time

log = logging.getLogger('valterra')
logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s')

ORDER_MWH    = 10
COMM_FIXED   = 20
COMM_PCT     = 0.0003
EXIT_Z       = 0.3
LOSS_STOP    = 60_000
DAILY_LIMIT  = 100_000
WEEKLY_LIMIT = 300_000
MIN_BARS     = 10
DB_PATH      = 'valterra.db'

state = {
    'today_spreads': [], 'open_position': False,
    'direction': 0, 'entry_spread': 0.0, 'entry_z': 0.0,
    'entry_bar': 0, 'bar_count': 0, 'daily_pnl': 0.0,
    'z_entry': 1.5, 'max_hold': 8, 'prime_hours': {17,18,19,20},
}
morning calibration
def morning_calibration():
    state['today_spreads'] = []; state['open_position'] = False
    state['bar_count'] = 0; state['daily_pnl'] = 0.0
    spreads_20d = db_get_recent_spreads(days=20)
    if len(spreads_20d) < 192:
        log.warning("Insufficient history — using defaults"); return
    A = float(np.std(spreads_20d))
    z_entry = 2.0 if A > 1400 else 1.5
    if datetime.today().weekday() == 6: z_entry += 0.5
    s = pd.Series(spreads_20d)
    B = float(s.autocorr(lag=8))
    if   B < 0:    max_hold = 6
    elif B < 0.10: max_hold = 4
    elif B < 0.25: max_hold = 8
    elif B < 0.35: max_hold = 10
    elif B < 0.45: max_hold = 12
    else:           max_hold = 16
    spreads_10d = db_get_recent_spreads(days=10)
    C = compute_peak_z_hour(spreads_10d)
    morning = {6,7,8,9} if C in {7,8} else set()
    evening = {17,18,19,20} if C < 20 else {18,19,20,21,22,23}
    state['z_entry'] = z_entry; state['max_hold'] = max_hold
    state['prime_hours'] = morning | evening
    log.info(f"Params: std={A:.0f} ac8={B:.3f} peak_hr={C} → z≥{z_entry} hold={max_hold}bars")
15-minute trading loop
def run_every_15_minutes():
    now = datetime.now(); hour = now.hour
    dam = get_dam_mcp(); rtm = get_rtm_mcp()
    if dam is None or rtm is None: return
    spread = dam - rtm
    state['today_spreads'].append(spread)
    state['bar_count'] += 1
    db_insert_bar(now, dam, rtm, spread)
    if len(state['today_spreads']) < MIN_BARS: return
    arr = np.array(state['today_spreads'])
    sigma = arr.std()
    z = (spread - arr.mean()) / sigma if sigma > 0 else 0.0
    if state['daily_pnl'] < -DAILY_LIMIT:
        if state['open_position']: close_position(spread, dam, rtm, 'daily_limit')
        return
    if hour >= 23 and now.minute >= 30:
        if state['open_position']: close_position(spread, dam, rtm, 'session_close')
        return
    if state['open_position']:
        direction = state['direction']
        hold = state['bar_count'] - state['entry_bar']
        unreal = direction * (state['entry_spread'] - spread) * ORDER_MWH
        if   unreal < -LOSS_STOP:                                  close_position(spread, dam, rtm, 'loss_stop')
        elif (direction==1 and z < EXIT_Z) or \
             (direction==-1 and z > -EXIT_Z):                   close_position(spread, dam, rtm, 'mean_revert')
        elif hold >= state['max_hold']:                          close_position(spread, dam, rtm, 'max_hold')
        return
    if hour not in state['prime_hours']: return
    if   z >  state['z_entry']: open_position( 1, spread, z, dam, rtm)
    elif z < -state['z_entry']: open_position(-1, spread, z, dam, rtm)
